<button onclick="tambahBahan()">+ Tambah</button>

<div id="sliders"></div>

<div class="result" id="output"></div>
</div>

<script>
let bahan = [
  { nama:"Mangga", value:30 },
  { nama:"Vanilla", value:30 },
  { nama:"Fresh", value:40 }
];

function tambahBahan(){
  let nama = document.getElementById("namaBahan").value;
  if(!nama) return;
  bahan.push({ nama:nama, value:20 });
  document.getElementById("namaBahan").value="";
  render();
}

function hapusBahan(i){
  bahan.splice(i,1);
  render();
}

function ubah(i,val){
  bahan[i].value = val;
  update();
}

function render(){
  let html="";
  bahan.forEach((b,i)=>{
    html += `
    <div class="slider-box">
      ${b.nama}: <span id="val${i}">${b.value}</span>%
      <input type="range" min="0" max="100" value="${b.value}"
      oninput="ubah(${i},this.value)">
      <button class="delete" onclick="hapusBahan(${i})">X</button>
    </div>`;
  });
  document.getElementById("sliders").innerHTML=html;
  update();
}

function analisis(){
  let sweet=0, fresh=0, woody=0;

  bahan.forEach(b=>{
    let nama = b.nama.toLowerCase();

    if(nama.includes("vanilla") || nama.includes("mangga") || nama.includes("peach")){
      sweet += Number(b.value);
    }
    if(nama.includes("fresh") || nama.includes("citrus") || nama.includes("blue")){
      fresh += Number(b.value);
    }
    if(nama.includes("woody") || nama.includes("hinoki") || nama.includes("musk")){
      woody += Number(b.value);
    }
  });

  let karakter="";
  if(sweet > fresh && sweet > woody) karakter="Sweet";
  else if(fresh > sweet && fresh > woody) karakter="Fresh";
  else karakter="Woody";

  let gender="Unisex";
  if(sweet > 60) gender="Cenderung Wanita";
  if(woody > 60) gender="Cenderung Pria";

  let tahan="Sedang";
  if(woody > 50) tahan="Tahan Lama";
  if(fresh > 60) tahan="Ringan";

  let saran="";
  if(sweet > 70) saran="Terlalu manis, tambahkan fresh/woody";
  else if(fresh > 70) saran="Terlalu ringan, tambahkan vanilla/musk";
  else saran="Komposisi sudah cukup balance";

  return {karakter, gender, tahan, saran};
}

function update(){
  let totalML = 30;
  let sum = bahan.reduce((a,b)=>a+Number(b.value),0);

  let hasil="<b>Hasil Racikan:</b><br>";

  bahan.forEach((b,i)=>{
    let ml=(b.value/sum)*totalML || 0;

    if(document.getElementById("val"+i)){
      document.getElementById("val"+i).innerText=b.value;
    }

    hasil += `${b.nama}: ${ml.toFixed(1)} ml<br>`;
  });

  let a = analisis();

  hasil += `<br><b>Analisis:</b><br>
  Karakter: ${a.karakter}<br>
  Gender: ${a.gender}<br>
  Ketahanan: ${a.tahan}<br>
  Saran: ${a.saran}`;

  document.getElementById("output").innerHTML=hasil;
}

render();
